-- =========================================================
-- CORPORATE HELP DESK / PostgreSQL schema
-- MVP version for Spring Boot + PostgreSQL
-- =========================================================

-- =========================================================
-- 1. СПРАВОЧНИКИ
-- =========================================================

CREATE TABLE roles (
    id              BIGSERIAL PRIMARY KEY,
    code            VARCHAR(50) NOT NULL UNIQUE,   -- INITIATOR, DISPATCHER, EXECUTOR
    name            VARCHAR(100) NOT NULL UNIQUE,  -- Инициатор, Диспетчер, Исполнитель
);

CREATE TABLE ticket_types (
    id                  BIGSERIAL PRIMARY KEY,
    code                VARCHAR(50) NOT NULL UNIQUE,   -- CONSULTATION, SERVICE_REQUEST, INCIDENT
    name                VARCHAR(100) NOT NULL UNIQUE,  -- Консультация, ЗНО, Инцидент
    default_sla_hours   INTEGER NOT NULL CHECK (default_sla_hours > 0),
);

CREATE TABLE work_groups (
    id              BIGSERIAL PRIMARY KEY,
    code            VARCHAR(100) NOT NULL UNIQUE,  -- help.desk.dispetcher, help.desk.service
    name            VARCHAR(255) NOT NULL UNIQUE,
    description     TEXT,
);

CREATE TABLE sections (
    id              BIGSERIAL PRIMARY KEY,
    code            VARCHAR(100) NOT NULL UNIQUE,
    name            VARCHAR(255) NOT NULL UNIQUE,
    description     TEXT,
    
);

CREATE TABLE service_directions (
    id                          BIGSERIAL PRIMARY KEY,
    section_id                  BIGINT NOT NULL REFERENCES sections(id) ON DELETE RESTRICT,
    code                        VARCHAR(100) NOT NULL UNIQUE,
    name                        VARCHAR(255) NOT NULL,
    description                 TEXT,
    default_ticket_type_id      BIGINT NOT NULL REFERENCES ticket_types(id) ON DELETE RESTRICT,
    CONSTRAINT uq_service_direction_name_per_section UNIQUE (section_id, name)
);

-- =========================================================
-- 2. ПОЛЬЗОВАТЕЛИ И АВТОРИЗАЦИЯ
-- =========================================================

CREATE TABLE users (
    id                  BIGSERIAL PRIMARY KEY,
    username            VARCHAR(100) NOT NULL UNIQUE,
    password_hash       VARCHAR(255) NOT NULL,
    email               VARCHAR(255) NOT NULL UNIQUE,
    last_name           VARCHAR(100) NOT NULL,
    first_name          VARCHAR(100) NOT NULL,
    middle_name         VARCHAR(100),
    role_id             BIGINT NOT NULL REFERENCES roles(id) ON DELETE RESTRICT,
    created_at          TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
);

-- Many-to-many: пользователь <-> рабочая группа
CREATE TABLE user_work_groups (
    user_id             BIGINT NOT NULL REFERENCES users(id) ON DELETE CASCADE,
    work_group_id       BIGINT NOT NULL REFERENCES work_groups(id) ON DELETE CASCADE,
    created_at          TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    PRIMARY KEY (user_id, work_group_id)
);

-- Many-to-many: направление обслуживания <-> рабочая группа
-- Нужна для хранения рекомендуемых РГ для выбранного направления
CREATE TABLE service_direction_work_groups (
    service_direction_id    BIGINT NOT NULL REFERENCES service_directions(id) ON DELETE CASCADE,
    work_group_id           BIGINT NOT NULL REFERENCES work_groups(id) ON DELETE CASCADE,
    is_default              BOOLEAN NOT NULL DEFAULT FALSE,
    created_at              TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    PRIMARY KEY (service_direction_id, work_group_id)
);

-- В одном направлении не более одной "рекомендуемой по умолчанию" группы
CREATE UNIQUE INDEX uq_service_direction_default_group
    ON service_direction_work_groups(service_direction_id)
    WHERE is_default = TRUE;

-- =========================================================
-- 3. ОБРАЩЕНИЯ
-- =========================================================

CREATE TABLE tickets (
    id                      BIGSERIAL PRIMARY KEY,

    -- Номер обращения, удобно показывать на UI
    ticket_number           VARCHAR(50) NOT NULL UNIQUE,

    initiator_id            BIGINT NOT NULL REFERENCES users(id) ON DELETE RESTRICT,
    section_id              BIGINT NOT NULL REFERENCES sections(id) ON DELETE RESTRICT,
    service_direction_id    BIGINT NOT NULL REFERENCES service_directions(id) ON DELETE RESTRICT,
    ticket_type_id          BIGINT NOT NULL REFERENCES ticket_types(id) ON DELETE RESTRICT,

    status                  VARCHAR(50) NOT NULL,
    work_group_id           BIGINT NOT NULL REFERENCES work_groups(id) ON DELETE RESTRICT,
    assignee_id             BIGINT REFERENCES users(id) ON DELETE RESTRICT,

    description             TEXT NOT NULL,

    created_at              TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    updated_at              TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
    resolved_at             TIMESTAMP NULL,
    deadline                TIMESTAMP NOT NULL,

    CONSTRAINT chk_ticket_status
        CHECK (status IN ('NEW', 'IN_PROGRESS', 'UNASSIGNED', 'RESOLVED')),

    -- UNASSIGNED допустим только у группы диспетчеров.
    -- Это правило лучше проверять в backend, потому что CHECK не умеет проверять code из другой таблицы.
    -- Здесь оставляем комментарий, а не кривой SQL-хак.

    -- resolved_at должен быть заполнен только у RESOLVED
    CONSTRAINT chk_ticket_resolved_at
        CHECK (
            (status = 'RESOLVED' AND resolved_at IS NOT NULL)
            OR
            (status <> 'RESOLVED')
        )
);

-- Индексы под типичные выборки
CREATE INDEX idx_tickets_initiator_id ON tickets(initiator_id);
CREATE INDEX idx_tickets_work_group_id ON tickets(work_group_id);
CREATE INDEX idx_tickets_assignee_id ON tickets(assignee_id);
CREATE INDEX idx_tickets_status ON tickets(status);
CREATE INDEX idx_tickets_created_at ON tickets(created_at DESC);
CREATE INDEX idx_tickets_deadline ON tickets(deadline);

-- =========================================================
-- 4. КОММЕНТАРИИ
-- =========================================================

CREATE TABLE comments (
    id                  BIGSERIAL PRIMARY KEY,
    ticket_id           BIGINT NOT NULL REFERENCES tickets(id) ON DELETE CASCADE,
    author_id           BIGINT NOT NULL REFERENCES users(id) ON DELETE RESTRICT,
    content             TEXT NOT NULL,
    is_private          BOOLEAN NOT NULL DEFAULT FALSE,
    created_at          TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
);

CREATE INDEX idx_comments_ticket_id ON comments(ticket_id);
CREATE INDEX idx_comments_author_id ON comments(author_id);
CREATE INDEX idx_comments_created_at ON comments(created_at);

-- =========================================================
-- 5. ЛЕНТА СОБЫТИЙ / ЖУРНАЛ
-- =========================================================

CREATE TABLE ticket_events (
    id                  BIGSERIAL PRIMARY KEY,
    ticket_id           BIGINT NOT NULL REFERENCES tickets(id) ON DELETE CASCADE,
    author_id           BIGINT NOT NULL REFERENCES users(id) ON DELETE RESTRICT,

    event_type          VARCHAR(50) NOT NULL,
    event_message       TEXT NOT NULL,

    created_at          TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,

    CONSTRAINT chk_ticket_event_type
        CHECK (event_type IN (
            'CREATED',
            'STATUS_CHANGED',
            'GROUP_ASSIGNED',
            'COMMENT_ADDED',
            'PRIVATE_COMMENT_ADDED',
            'TYPE_CHANGED',
            'SLA_RECALCULATED',
            'EMPLOYEE_ASSIGNED'
        ))
);

CREATE INDEX idx_ticket_events_ticket_id ON ticket_events(ticket_id);
CREATE INDEX idx_ticket_events_created_at ON ticket_events(created_at);
CREATE INDEX idx_ticket_events_event_type ON ticket_events(event_type);

-- =========================================================
-- 6. УВЕДОМЛЕНИЯ
-- =========================================================

CREATE TABLE notifications (
    id                  BIGSERIAL PRIMARY KEY,
    user_id             BIGINT NOT NULL REFERENCES users(id) ON DELETE CASCADE,
    ticket_id           BIGINT NOT NULL REFERENCES tickets(id) ON DELETE CASCADE,

    title               VARCHAR(255) NOT NULL,
    message             TEXT NOT NULL,

    is_read             BOOLEAN NOT NULL DEFAULT FALSE,
    
    created_at          TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
);

CREATE INDEX idx_notifications_user_id ON notifications(user_id);
CREATE INDEX idx_notifications_ticket_id ON notifications(ticket_id);
CREATE INDEX idx_notifications_is_read ON notifications(is_read);
CREATE INDEX idx_notifications_created_at ON notifications(created_at DESC);

-- =========================================================
-- 8. ПОЛЕЗНЫЕ VIEW ДЛЯ ЧТЕНИЯ
-- =========================================================

CREATE OR REPLACE VIEW v_ticket_full AS
SELECT
    t.id,
    t.ticket_number,
    t.status,
    t.description,
    t.created_at,
    t.updated_at,
    t.deadline,
    t.resolved_at,

    i.username AS initiator_username,
    i.last_name || ' ' || i.first_name || COALESCE(' ' || i.middle_name, '') AS initiator_fio,
    i.email AS initiator_email,

    s.name AS section_name,
    sd.name AS service_direction_name,
    tt.name AS ticket_type_name,
    tt.default_sla_hours,

    wg.code AS work_group_code,
    wg.name AS work_group_name,

    a.username AS assignee_username,
    a.last_name || ' ' || a.first_name || COALESCE(' ' || a.middle_name, '') AS assignee_fio

FROM tickets t
JOIN users i ON i.id = t.initiator_id
JOIN sections s ON s.id = t.section_id
JOIN service_directions sd ON sd.id = t.service_direction_id
JOIN ticket_types tt ON tt.id = t.ticket_type_id
JOIN work_groups wg ON wg.id = t.work_group_id
LEFT JOIN users a ON a.id = t.assignee_id;
